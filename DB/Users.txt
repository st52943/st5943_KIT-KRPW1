CREATE TABLE IF NOT EXISTS Users (
    ID INTEGER NOT NULL,
    UsersRole_ID INTEGER NOT NULL,
    username VARCHAR(100),
    password VARCHAR(100),
    license VARCHAR(100),
    PRIMARY KEY (ID)
);

CREATE TABLE IF NOT EXISTS UsersRole (
    ID INTEGER NOT NULL,
    Role VARCHAR(100),
    PRIMARY KEY (ID)
);

ALTER TABLE Users
    ADD CONSTRAINT Users_UsersRole_FK FOREIGN KEY (UsersRole_ID)
    REFERENCES UsersRole(ID)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION;


ALTER TABLE Users
MODIFY COLUMN ID INTEGER AUTO_INCREMENT;
